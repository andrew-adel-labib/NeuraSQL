Sales Before Previous Month QTY = 
VAR SelMaxDate =
    MAX('DateDim'[Date])

VAR TodayOrSel =
    MIN(SelMaxDate, TODAY())

-- Current month start
VAR CurrentStartDate =
    CALCULATE(
        MIN('DateDim'[Date]),
        FILTER(
            ALL('DateDim'),
            YEAR('DateDim'[Date]) = YEAR(TodayOrSel)
            && MONTH('DateDim'[Date]) = MONTH(TodayOrSel)
        )
    )

-- Is current month complete?
VAR IsMonthComplete =
    TodayOrSel = EOMONTH(TodayOrSel, 0)

-- Previous month start
VAR PreviousStartDate =
    CALCULATE(
        MIN('DateDim'[Date]),
        FILTER(
            ALL('DateDim'),
            YEAR('DateDim'[Date]) = YEAR( EDATE(CurrentStartDate, -1) )
            && MONTH('DateDim'[Date]) = MONTH( EDATE(CurrentStartDate, -2) )
        )
    )

-- If incomplete: same day in previous month (safe)
VAR SameDayPrevMonth =
VAR TargetDay = DAY(TodayOrSel)
VAR PrevYear  = YEAR(PreviousStartDate)
VAR PrevMonth = MONTH(PreviousStartDate)
VAR Try =
    DATE(PrevYear, PrevMonth, TargetDay)
RETURN
    IF(
        MONTH(Try) = PrevMonth,
        Try,
        EOMONTH(TodayOrSel, -2)
    )

-- Final PreviousEndDate:
VAR PreviousEndDate =
    IF(
        IsMonthComplete,
        EOMONTH(TodayOrSel, -2),     -- full previous month
        SameDayPrevMonth             -- same day logic
    )

-- Calculate Qty
VAR Prev =
    CALCULATE(
        [Sales QTY Ton],
        DATESBETWEEN(
            'DateDim'[Date],
            PreviousStartDate,
            PreviousEndDate
        )
    )

RETURN IF(Prev = 0, 0, Prev)
